const fs = require("fs")
const path = require("path")

const FCC_PATH = "./freeCodeCamp/curriculum/challenges/english"

function transformFCC() {
  const superBlocks = fs.readdirSync(FCC_PATH)

  for (const blockFolder of superBlocks) {
    const blockPath = path.join(FCC_PATH, blockFolder)

    const meta = JSON.parse(
      fs.readFileSync(path.join(blockPath, "_meta.json"))
    )

    const course = {
      name: meta.title,
      description: "Imported from FCC",
      level: "BEGINNER",
      modules: []
    }

    const files = fs.readdirSync(blockPath)

    files.forEach(file => {
      if (file.endsWith(".json") && file !== "_meta.json") {
        const challenge = JSON.parse(
          fs.readFileSync(path.join(blockPath, file))
        )

        // group by challenge.block (module)
      }
    })

    fs.writeFileSync(
      `./seed-data/fcc/${meta.name}.json`,
      JSON.stringify(course, null, 2)
    )
  }
}

üöÄ PHASE 5 ‚Äî Seed Your Prisma DB

Example:

import { PrismaClient } from "@prisma/client"
import fs from "fs"

const prisma = new PrismaClient()

async function main() {
  const courseData = JSON.parse(
    fs.readFileSync("./seed-data/fcc/responsive-web-design.json", "utf-8")
  )

  const course = await prisma.course.create({
    data: {
      name: courseData.name,
      description: courseData.description,
      level: "BEGINNER",
      modules: {
        create: courseData.modules.map(module => ({
          name: module.name,
          orderIndex: module.orderIndex,
          description: "",
          problems: {
            create: module.problems.map(problem => ({
              title: problem.title,
              description: problem.description,
              type: problem.type,
              difficulty: problem.difficulty
            }))
          }
        }))
      }
    }
  })
}

main()
